75  MiaTransmissionHandle(const std::string& name,
76  transmission_interface::MiaIndexTransmission* transmission,
77  const ActuatorData& actuator_data,
78  const JointData& joint_data,
79  const ActuatorData& actuator_state)
80  : name_(name),
81  transmission_(transmission),
82  actuator_data_(actuator_data),
83  joint_data_(joint_data),
84  actuator_state_(actuator_state)
85  {
86  // Precondition: Valid transmission
87  if (!transmission_)
88  {
89  throw TransmissionInterfaceException("Unspecified transmission.");
90  }
91 
92  // Catch trivial error: All data vectors are empty (handle can't do anything without data)
93  if (actuator_data.position.empty() && actuator_data.velocity.empty() && actuator_data.effort.empty() &&
94  joint_data.position.empty() && joint_data.velocity.empty() && joint_data.effort.empty() &&
95  actuator_state.position.empty() && actuator_state.velocity.empty() && actuator_state.effort.empty())
96  {
97  throw TransmissionInterfaceException("All data vectors are empty. Transmission instance can't do anything!.");
98  }
99 
100  // Precondition: All non-empty data vectors must have sizes consistent with the transmission
101  if (!actuator_data.position.empty() && actuator_data.position.size() != transmission_->numActuators() )
102  {
103  throw TransmissionInterfaceException("Actuator position data size does not match transmission.");
104  }
105  if (!actuator_data.velocity.empty() && actuator_data.velocity.size() != transmission_->numActuators())
106  {
107  throw TransmissionInterfaceException("Actuator velocity data size does not match transmission.");
108  }
109  if (!actuator_data.effort.empty() && actuator_data.effort.size() != transmission_->numActuators())
110  {
111  throw TransmissionInterfaceException("Actuator effort data size does not match transmission.");
112  }
113 
114  if (!joint_data.position.empty() && joint_data.position.size() != transmission_->numJoints())
115  {
116  throw TransmissionInterfaceException("Joint position data size does not match transmission.");
117  }
118  if (!joint_data.velocity.empty() && joint_data.velocity.size() != transmission_->numJoints())
119  {
120  throw TransmissionInterfaceException("Joint velocity data size does not match transmission.");
121  }
122  if (!joint_data.effort.empty() && joint_data.effort.size() != transmission_->numJoints())
123  {
124  throw TransmissionInterfaceException("Joint effort data size does not match transmission.");
125  }
126 
127  // Added
128  if (!actuator_state.position.empty() && actuator_state.position.size() != transmission_->numActuators() )
129  {
130  throw TransmissionInterfaceException("Actuator position state size does not match transmission.");
131  }
132  if (!actuator_state.velocity.empty() && actuator_state.velocity.size() != transmission_->numActuators())
133  {
134  throw TransmissionInterfaceException("Actuator velocity state size does not match transmission.");
135  }
136  if (!actuator_state.effort.empty() && actuator_state.effort.size() != transmission_->numActuators())
137  {
138  throw TransmissionInterfaceException("Actuator effort state size does not match transmission.");
139  }
141 
142  // Precondition: Valid pointers to raw data
143  if (!hasValidPointers(actuator_data.position))
144  {
145  throw TransmissionInterfaceException("Actuator position data contains null pointers.");
146  }
147  if (!hasValidPointers(actuator_data.velocity))
148  {
149  throw TransmissionInterfaceException("Actuator velocity data contains null pointers.");
150  }
151  if (!hasValidPointers(actuator_data.effort))
152  {
153  throw TransmissionInterfaceException("Actuator effort data contains null pointers.");
154  }
155 
156  if (!hasValidPointers(joint_data.position))
157  {
158  throw TransmissionInterfaceException("Joint position data contains null pointers.");
159  }
160  if (!hasValidPointers(joint_data.velocity))
161  {
162  throw TransmissionInterfaceException("Joint velocity data contains null pointers.");
163  }
164  if (!hasValidPointers(joint_data.effort))
165  {
166  throw TransmissionInterfaceException("Joint effort data contains null pointers.");
167  }
168 
169  // Added
170  if (!hasValidPointers(actuator_state.position))
171  {
172  throw TransmissionInterfaceException("Actuator position state contains null pointers.");
173  }
174  if (!hasValidPointers(actuator_state.velocity))
175  {
176  throw TransmissionInterfaceException("Actuator velocity state contains null pointers.");
177  }
178  if (!hasValidPointers(actuator_state.effort))
179  {
180  throw TransmissionInterfaceException("Actuator effort state contains null pointers.");
181  }
182  }
183 
184  private:
185  static bool hasValidPointers(const std::vector<double*>& data)
186  {
187  for (std::vector<double*>::const_iterator it = data.begin(); it != data.end(); ++it)
188  {
189  if (!(*it)) {return false;}
190  }
191  return true;
192  }
193  };

378  template <class HandleType>
379  class MiaTransmissionInterface : public hardware_interface::ResourceManager<HandleType>
380  {
381  public:
382 
383  HandleType getHandle(const std::string& name)
384  {
385  // Rethrow exception with a meaningful type
386  try
387  {
388  return this->hardware_interface::ResourceManager<HandleType>::getHandle(name);
389  }
390  catch(const std::logic_error& e)
391  {
392  throw TransmissionInterfaceException(e.what());
393  }
394  }
395 
396  void propagate()
397  {
398  typedef typename hardware_interface::ResourceManager<HandleType>::ResourceMap::iterator ItratorType;
399  for (ItratorType it = this->resource_map_.begin(); it != this->resource_map_.end(); ++it)
400  {
401  it->second.propagate();
402  }
403  }
404  /*\}*/
405  };
