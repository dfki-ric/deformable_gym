82  double thumb_opp_passive_joint::GetThumbOppPosition(double j_index_flex_pos)
83  {
84  // evaluate scale and offset
85  //Th_opp_scale_ = (y(1) - y(2))/(x(1)-x(2));
86  // Th_opp_offset = y(1) - (Th_abd_scale*x(1));
87 
88  //double empirical_scale_j_thumb_opp_pos = 0.86418; // 0.95238; old with cmd 25
89  //double empirical_scale_j_thumb_opp_pos = -0.063146;// -0.0052884;old with cmd 25
90  double j_index_flex_pos_filt = (floor(j_index_flex_pos*1000))/1000; // filt the j_index_fle pose to avoid shakin
91  th_opp_scale_ = (th_min_pos_ - th_max_pos_) / (th_opp_start_close_index_angle_ - th_opp_stop_close_index_angle_);
92  th_opp_offset_ = th_min_pos_ - (th_opp_scale_*th_opp_start_close_index_angle_ );
93  //ROS_INFO_STREAM("thumb_opp_passive_joint: "<< "Scale: "<< th_opp_scale_<< " Offset: "<<th_opp_offset_);
94 
95  // Implementation of g3(delta) according with Mia trasmission and empirical test
96  double j_th_opp_target_position_temp = th_opp_scale_ * (j_index_flex_pos_filt) + th_opp_offset_;
97  double j_th_opp_target_position;
98 
99 
100  if (j_th_opp_target_position_temp <= th_min_pos_)
101  j_th_opp_target_position = th_min_pos_ + 0.02;
102 
103  else if (j_th_opp_target_position_temp >= th_max_pos_)
104  j_th_opp_target_position = th_max_pos_ - 0.02;
105 
106  else
107  j_th_opp_target_position = j_th_opp_target_position_temp;
108 
109  //ROS_INFO_STREAM("Scale: "<< th_opp_scale_<< " Offset: "<<th_opp_offset_ << " Ix_: "<< j_index_flex_pos<< " Ix_f: "<< j_index_flex_pos_filt<< " thOpp :"<< j_th_opp_target_position_temp << "->" << j_th_opp_target_position);
110 
111  return j_th_opp_target_position;
112  }

116  bool thumb_opp_passive_joint::LoadURDFInfo()
117  {
118  bool result = false;
119 
120  const std::string urdf_string = getURDF(robot_description_);
121 
122  // get URDF model
123  urdf::Model urdf_model;
124  const urdf::Model *const urdf_model_ptr = urdf_model.initString(urdf_string) ? &urdf_model : NULL;
125 
126  const urdf::JointConstSharedPtr urdf_joint = urdf_model_ptr->getJoint(j_thumb_name);
127 
128  joint_limits_interface::JointLimits limits;
129 
130  if (urdf_joint != NULL)
131  {
132 
133  // Get limits from the URDF file.
134  if (joint_limits_interface::getJointLimits(urdf_joint, limits))
135  {
136  updateThOppJointLimits(limits.min_position, limits.max_position);
137 
138  result = true;
139  }
140  }
141  else
142  {
143  updateThOppJointLimits(0.0, 0.0);
144  result = false;
145  }
146 
147  return result;
148  }

159  std::string thumb_opp_passive_joint::getURDF(std::string param_name) const
160  {
161  std::string urdf_string;
162 
163  // search and wait for robot_description on param server
164  while (urdf_string.empty())
165  {
166  std::string search_param_name;
167 
168  if (n.searchParam(param_name, search_param_name))
169  {
170  ROS_INFO_ONCE_NAMED("remap_mia_joint_states", "node is waiting for model"
171  " URDF in parameter on the ROS param server.");
172 
173  n.getParam(search_param_name, urdf_string);
174  }
175  else
176  {
177  ROS_INFO_ONCE_NAMED("remap_mia_joint_states", "Node is waiting for model"
178  " URDF in parameter on the ROS param server.");
179 
180  n.getParam(param_name, urdf_string);
181  }
182 
183  usleep(100000);
184  }
185  ROS_DEBUG_STREAM_NAMED("remap_mia_joint_states", "Recieved urdf from param server, parsing...");
186  return urdf_string;
187  }
