111 // mu = h_thfle(pos)
112 inline double MiaThfleTransmission::h_thfle(const double pos)
113 {
114  double scale = h_tf_scale_;
115  double offset = h_tf_offset_;
116  double mu = scale * pos + offset;
117  return mu;
118 }
119 
120 // pos = h_thfle_inv(mu)
121 double MiaThfleTransmission::h_thfle_inv(const double mu)
122 {
123  double scale = h_tf_inv_scale_;
124  double offset = h_tf_inv_offset_;
125  double pos = (double) round( scale * mu + offset); // cmd
126  return pos;
127 }

170 void MiaThfleTransmission::actuatorToJointVelocity(const ActuatorData& act_data,
171  JointData& jnt_data)
172 {
173 
174  double spe = *act_data.velocity[0];
175  double omega_m = dh(spe);
176  double omega_f = omega_m / reduction_;
177 
178  *jnt_data.velocity[0] = omega_f;
