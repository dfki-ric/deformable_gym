129  void chattercallback(const sensor_msgs::JointState::ConstPtr& msg)
130  {
131  int nJoints = msg->name.size();
132 
133  sensor_msgs::JointState sensor_msgs = *msg;
134 
135  int j_index = -1;
136  int j_th = -1;
137 
138  for(unsigned int j=0; j < nJoints; j++)
139  {
140  const std::string jname = sensor_msgs.name[j];
141  if (jname == j_index_name) {j_index = j;}
142  if (jname == j_thumb_name) {j_th = j;}
143  if(j_index > -1 && j_th > -1) {break;}
144  }
145 
146  // Remap
147 
148 
149  if (j_index > -1 && j_th >-1 && remap_th_opp == true)
150  {
151  double th_fle_pos = MyTh_opp_passiveJoint.GetThumbOppPosition(sensor_msgs.position[j_index]);
152  sensor_msgs.position[j_th] = th_fle_pos;
153  sensor_msgs.position[j_index] = std::abs(sensor_msgs.position[j_index]);
154 
155  }
156  else if (j_index > -1 && (j_th == -1 || remap_th_opp == false))
157  {
158  sensor_msgs.position[j_index] = std::abs(sensor_msgs.position[j_index]);
159  }
160 
161  pub.publish(sensor_msgs);
162 
163  }
