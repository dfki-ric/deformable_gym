381  // RegisterJointLimits:Register the limits of the joint specified by joint_name and joint_handle.
382  // The limits are retrieved from joint_limit_nh. If urdf_model is not NULL, limits are retrieved from it also.
383  // Return the joint's type, lower position limit, upper position limit, and effort limit.
385  void MiaHWSim::registerJointLimits(const std::string& joint_name,
386  const std::vector<hardware_interface::JointHandle>& joint_handle,
387  const std::vector<ControlMethod> ctrl_method,
388  const ros::NodeHandle& joint_limit_nh,
389  const urdf::Model *const urdf_model,
390  int *const joint_type, double *const lower_limit,
391  double *const upper_limit, double *const effort_limit)
392  {
393  // Initialize variables
394  *joint_type = urdf::Joint::UNKNOWN;
395  *lower_limit = -std::numeric_limits<double>::max();
396  *upper_limit = std::numeric_limits<double>::max();
397  *effort_limit = std::numeric_limits<double>::max();
398 
399  joint_limits_interface::JointLimits limits;
400  bool has_limits = false;
401  joint_limits_interface::SoftJointLimits soft_limits;
402  bool has_soft_limits = false;
403 
404  // find limits
405  if (urdf_model != NULL)
406  {
407  //const boost::shared_ptr<const urdf::Joint> urdf_joint = urdf_model->getJoint(joint_name);
408  const urdf::JointConstSharedPtr urdf_joint = urdf_model->getJoint(joint_name);
409  if (urdf_joint != NULL)
410  {
411  *joint_type = urdf_joint->type;
412  // Get limits from the URDF file.
413  if (joint_limits_interface::getJointLimits(urdf_joint, limits))
414  has_limits = true;
415  if (joint_limits_interface::getSoftJointLimits(urdf_joint, soft_limits))
416  has_soft_limits = true;
417  }
418  }
419  // Get limits from the parameter server.
420  if (joint_limits_interface::getJointLimits(joint_name, joint_limit_nh, limits))
421  has_limits = true;
422 
423  if (!has_limits)
424  return;
425 
426  if (*joint_type == urdf::Joint::UNKNOWN)
427  {
428  // Infer the joint type.
429 
430  if (limits.has_position_limits)
431  {
432  *joint_type = urdf::Joint::REVOLUTE;
433  }
434  else
435  {
436  if (limits.angle_wraparound)
437  *joint_type = urdf::Joint::CONTINUOUS;
438  else
439  *joint_type = urdf::Joint::PRISMATIC;
440  }
441  }
442 
443  if (limits.has_position_limits)
444  {
445  *lower_limit = limits.min_position;
446  *upper_limit = limits.max_position;
447  }
448  if (limits.has_effort_limits)
449  *effort_limit = limits.max_effort;
450 
451  if (has_soft_limits)
452  {
453  for (unsigned int cm = 0; cm< ctrl_method.size(); cm++)
454  {
455  switch (ctrl_method[cm])
456  {
457  case EFFORT:
458  {
459  const joint_limits_interface::EffortJointSoftLimitsHandle
460  limits_handle(joint_handle[cm], limits, soft_limits);
461  ej_limits_interface_.registerHandle(limits_handle);
462  }
463  break;
464  case POSITION:
465  {
466  const joint_limits_interface::PositionJointSoftLimitsHandle
467  limits_handle(joint_handle[cm], limits, soft_limits);
468  pj_limits_interface_.registerHandle(limits_handle);
469  }
470  break;
471  case VELOCITY:
472  {
473  const joint_limits_interface::VelocityJointSoftLimitsHandle
474  limits_handle(joint_handle[cm], limits, soft_limits);
475  vj_limits_interface_.registerHandle(limits_handle);
476  }
477  break;
478  }
479  }
480  }
481  else
482  {
483  for (unsigned int cm = 0; cm< ctrl_method.size(); cm++)
484  {
485  switch (ctrl_method[cm])
486  {
487 
488  case EFFORT:
489  {
490  const joint_limits_interface::EffortJointSaturationHandle
491  sat_handle(joint_handle[cm], limits);
492  ej_sat_interface_.registerHandle(sat_handle);
493  }
494  break;
495  case POSITION:
496  {
497  const joint_limits_interface::PositionJointSaturationHandle
498  sat_handle(joint_handle[cm], limits);
499  pj_sat_interface_.registerHandle(sat_handle);
500  }
501  break;
502  case VELOCITY:
503  {
504  const joint_limits_interface::VelocityJointSaturationHandle
505  sat_handle(joint_handle[cm], limits);
506  vj_sat_interface_.registerHandle(sat_handle);
507  }
508  break;
509 
510  }//endswicth
511  }
512 
513  }//endifelse
514 
515  }//end registerlimits function

519  //readSim: Read state data from simulated robot hardware such as joint position, velocities,effort </summury>
521  void MiaHWSim::readSim(ros::Time time, ros::Duration period)
522  {
523  for(unsigned int j=0; j < n_dof_; j++)
524  {
525  double position;
526  double velocity;
527  double effort;
528 
529  //*
530  // 1) Read position
531  if (joint_types_[j] == urdf::Joint::PRISMATIC)
532  {
533  #if GAZEBO_MAJOR_VERSION >= 9
534  position = sim_joints_[j]->Position (0); // used with Gazebo 9
535  #else
536 
537  position = sim_joints_[j]->GetAngle(0).Radian(); //Deprecated used for Gazebo 7.1.0,
538  #endif
539 
540  }
541  else
542  {
543  #if GAZEBO_MAJOR_VERSION >= 9
544  position = joint_position_[j] + angles::shortest_angular_distance(joint_position_[j],
545  sim_joints_[j]->Position (0));
546 
547  #else
548  position = joint_position_[j] + angles::shortest_angular_distance(joint_position_[j],
549  sim_joints_[j]->GetAngle(0).Radian());
550  #endif
551  }
552 
553  //*
554  // 2) Read velocity
555  if (j == joints_ii.j_thumb_opp || j == joints_ii.j_thumb_flex || j == joints_ii.j_index_flex ||
556  j == joints_ii.j_mrl_2 || j == joints_ii.j_mrl_3 || j== joints_ii.j_mrl_flex)
557  {
558  velocity = (position - joint_position_[j]) / period.toSec(); // rad/sec estimated for mia_hand
559  }
560  else
561  {
562  velocity = sim_joints_[j]->GetVelocity(0); // return wrong values
563  }
564  //*
565  // 3) Read effort
566  effort = sim_joints_[j]->GetForce((unsigned int)(0));
567 
568  //*
569  // 4) Publish joint states
570  if (j == joints_ii.j_index_flex)
571  {
572  // Read velocity with the right sign for the index
573  velocity = ((position * j_index_flex_sign) - joint_position_[j]) / period.toSec(); // rad/sec estimated
574 
575  joint_position_[j] = (position * j_index_flex_sign);
576  joint_velocity_[j] = velocity;
577  joint_effort_[j] = effort * j_index_flex_sign;
578  }
579  else
580  {
581 
582  joint_position_[j] = position;
583  joint_velocity_[j] = velocity;
584  joint_effort_[j] = effort;
585  }
586 
587  } // end for joint
588 
589  }// end readSim function

592  //writeSim: Write commands to the simulated robot hardware. such as joint position, velocities,
593  // effort commands
595  void MiaHWSim::writeSim(ros::Time time, ros::Duration period)
596  {
597 
598  /* Handle the initialization of the last commands when this function
599  has been called for the first time */
600  if(write_counter == 0)
601  {
602  // Initialize variables used to select the control methods for the joints
603  for(unsigned int j=0; j < n_dof_; j++)
604  {
605  // for the first iteration initialize the last command and the last joint position status
606  last_joint_position_command_[j] = joint_position_command_ [j];
607  last_joint_velocity_command_[j] = joint_velocity_command_[j];
608  last_joint_effort_command_[j] = joint_effort_command_[j];
609 
610  last_joint_control_methods_[j] = POSITION;
611  }
612  write_counter++;
613  }
614 
615  // If the E-stop is active, joints controlled by position commands will maintain their positions.
616  if (e_stop_active_)
617  {
618  if (!last_e_stop_active_)
619  {
620  last_joint_position_command_ = joint_position_;
621  last_e_stop_active_ = true;
622  }
623  joint_position_command_ = last_joint_position_command_;
624  }
625  else
626  {
627  last_e_stop_active_ = false;
628  }
629 
630  // Enforce limits
631  ej_sat_interface_.enforceLimits(period);
632  ej_limits_interface_.enforceLimits(period);
633  pj_sat_interface_.enforceLimits(period);
634  pj_limits_interface_.enforceLimits(period);
635  vj_sat_interface_.enforceLimits(period);
636  vj_limits_interface_.enforceLimits(period);
637 
638  // Set the max difference position that is allowed between middle, ring and middle finger
639  const double MaxDiff_mrl = 0.1; // rad
640 
641  for(unsigned int j=0; j < n_dof_; j++)
642  {
643 
644  /* Skip write action for j_mrl_2 and j_mrl_3 joints since they
645  have to follow j_mrl_flex joint */
