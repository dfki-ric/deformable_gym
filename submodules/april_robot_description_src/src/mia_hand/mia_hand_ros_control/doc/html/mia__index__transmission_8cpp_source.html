150  double MiaIndexTransmission::f(const double alpha )
151  {
152 
153  double delta;
154 
155  // input larger than the last x interval limit
156  if (alpha > x_intervals_f_[x_intervals_f_.size() - 1 ])
157  {
158  delta = f_scale_[x_intervals_f_.size() - 1] * alpha + f_offset_[x_intervals_f_.size() - 1];
159  }
160  else
161  {
162 
163  for(unsigned int j=0; j < x_intervals_f_.size() ; j++)
164  {
165 
166  // First interval
167  if (j == 0)
168  {
169  if (alpha <= x_intervals_f_[j])
170  {
171  delta = f_scale_[j] * alpha + f_offset_[j];
172  break;
173  }
174  }
175  else // Other intervals
176  {
177  if (alpha > x_intervals_f_[j-1] && alpha <= x_intervals_f_[j])
178  {
179  delta = f_scale_[j] * alpha + f_offset_[j];
180  break;
181  }
182  }
183  }
184 
185  }
186  return delta;

282  // delta_calib = f2(delta)
283  double MiaIndexTransmission::f2(const double delta)
284  {
285  double offset = f2_offset_;
286  double delta_calib = delta;
287 
288  if(offset > 0)
289  {
290  if(delta >= 0){
291  delta_calib = delta+offset;}
292  else {
293  delta_calib = delta - offset;}
294  }
295  else if (offset < 0)
296  {
297  if(delta >= abs(offset)){
298  delta_calib = delta + offset; // pull down
299  }
300  else if (delta <= -abs(offset)){
301  delta_calib = delta - offset; // pull up
302  }
303  else if (abs(delta) < abs(offset)) {
304  delta_calib = 0; // zeros
305  }
306  }
307 
308  //ROS_INFO_STREAM("MiaIndexTransmission: "<< "f2 delta: "<< delta <<" --> delta_calib: "<< delta_calib << " -- offset calib: "<<offset);
309  return delta_calib;
310  }

312  // delta = f2_inv(delta_calib)
313  double MiaIndexTransmission::f2_inv(const double delta_calib)
314  {
315  double offset = f2_offset_;
316  double delta = delta_calib;
317 
318  if(offset > 0)
319  {
320  if(delta_calib >= abs(offset)){
321  delta = delta_calib - offset;
322  }
323  else if (delta_calib <= -abs (offset)){
324  delta = delta_calib + offset;
325  }
326  else if (abs(delta_calib) < abs(offset)) {
327  delta = offset; // zeros
328  }
329  }
330  else if(offset < 0)
331  {
332  if(delta_calib >= 0){
333  delta = delta_calib - offset;
334  }
335  else if(delta_calib < 0){
336  delta = delta_calib + offset;
337  }
338  }
339 
340  return delta;
