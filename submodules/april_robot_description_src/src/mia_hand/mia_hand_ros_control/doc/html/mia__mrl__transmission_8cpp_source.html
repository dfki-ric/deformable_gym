111 // mu = h_mrl(pos)
112 inline double MiaMrlTransmission::h_mrl(const double pos)
113 {
114  double scale = h_mrl_scale_;
115  double offset = h_mrl_offset_;
116  double mu = scale * pos + offset;
117 
118  //ROS_INFO_STREAM("MiaMrlTransmission: "<< "pos: "<< pos <<" --> mu: "<< mu<<" -- Scale: "<< scale <<" offset: "<< offset );
119  return mu;
120 }

122 // pos = h_mrl_inv(mu)
123 double MiaMrlTransmission::h_mrl_inv(const double mu)
124 {
125  double scale = h_mrl_inv_scale_;
126  double offset = h_mrl_inv_offset_;
127  double pos = (double) round( scale * mu + offset); // cmd
128  return pos;
129 }

132 double MiaMrlTransmission::dh (const double spe)
133 {

172 void MiaMrlTransmission::actuatorToJointVelocity(const ActuatorData& act_data,
173  JointData& jnt_data)
174 {
175 
176  double spe = *act_data.velocity[0];
177  double omega_m = dh(spe);
178  double omega_f = omega_m / reduction_;
179 
180  *jnt_data.velocity[0] = omega_f;
181 }
