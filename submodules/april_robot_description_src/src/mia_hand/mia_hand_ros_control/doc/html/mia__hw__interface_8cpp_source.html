105  // Connect Mia
107  bool MiaHWInterface:: connect_mia(uint16_t port_num)

138  }

140  connectionmia_mtx_.unlock();

573  //read: Read actuator state from hardware and propagate to joint spce
575  void MiaHWInterface::read(const ros::Time& time, const ros::Duration& duration)
576  {
577  // Get from Mia hand
578  for (uint8_t jnt = 0; jnt < n_actuators_; jnt++)
579  {
580  act_position_state_[jnt] = (double)mia_.getMotorPos(jnt);
581  act_velocity_state_[jnt] = (double)mia_.getMotorSpe(jnt);
582 
583  if (read_counter== 0 )
584  {
585  old_act_position_state_[jnt] = act_position_state_[jnt] ;
586  read_counter ++ ;
587  }
588  else
589  {
590  if (act_position_state_[jnt] < old_act_position_state_[jnt] )
591  act_velocity_state_[jnt] = - act_velocity_state_[jnt];
592  old_act_position_state_[jnt] = act_position_state_[jnt] ;
593  }
594  }
595 
596 
597  // propagate state trasmission
598  act_to_jnt_pos_state.propagate();
599  act_to_jnt_vel_state.propagate();
600  index_act_to_jnt_pos_state.propagate();
601  index_act_to_jnt_vel_state.propagate();
602 
603  // update the sate of the other passive joints (helpfull for visualization tools)
604  joint_position_state_[joints_ii.j_mrl_2 ] = joint_position_state_[joints_ii.j_mrl_flex ];
605  joint_velocity_state_[joints_ii.j_mrl_2 ] = joint_velocity_state_[joints_ii.j_mrl_flex ];
606 
607  joint_position_state_[joints_ii.j_mrl_3 ] = joint_position_state_[joints_ii.j_mrl_flex ];
608  joint_velocity_state_[joints_ii.j_mrl_3 ] = joint_velocity_state_[joints_ii.j_mrl_flex ];
609 
610  joint_position_state_[joints_ii.j_thumb_opp ] = GetThumbOppPosition();
611  joint_velocity_state_[joints_ii.j_thumb_opp ] = 0;
612 
613  // publish current stream if is_fin_for_info_enabled
614  if(is_fin_for_info_enabled)
615  {
616  mia_hand_msgs::FingersStrainGauges msg_sg;
617 
618  mia_.getFingerSgRaw(0, msg_sg.thu[0], msg_sg.thu[1]);
619  mia_.getFingerSgRaw(2, msg_sg.ind[0], msg_sg.ind[1]);
620  mia_.getFingerSgRaw(1, msg_sg.mrl[0], msg_sg.mrl[1]);
621 
622  fin_for_info_.publish(msg_sg);
623  }
624 
625  if(is_mot_cur_info_enabled)
626  {
627  mia_hand_msgs::FingersData msg;
628 
629  msg.thu = mia_.getMotorCur(0);
630  msg.ind = mia_.getMotorCur(2);
631  msg.mrl = mia_.getMotorCur(1);
632 
633  mot_cur_info_.publish(msg);
634  }
635 
636  // publish the hand connection ComponentStatus
637  bool status_ = checkConnection();
638  mia_hand_msgs::ComponentStatus msg_status;
639  msg_status.status = status_;
640  if(status_)
641  {
642  msg_status.msg = "Mia hand connected";
643  }
644  else
645  {
646  msg_status.msg = "Mia hand disconnected";
647  }
648  connection_status_info_.publish(msg_status);
649 
650  return;
651  }

654  //write: Get joint target from ros interfaces, propagate to the actuator space and
655  // Write the actuator target to the hardware.
657  void MiaHWInterface::write(const ros::Time& time, const ros::Duration& duration)
658  {
659  /* Selectively send position or speed command according to
660  * the external commands sent by the controllers.
661  */
662  if(write_counter == 0)
663  {

746  void MiaHWInterface::registerJointLimits(const std::string& joint_name,
747  const std::vector<hardware_interface::JointHandle>& joint_handle,

756  *lower_limit = -std::numeric_limits<double>::max();
757  *upper_limit = std::numeric_limits<double>::max();
758  *effort_limit = std::numeric_limits<double>::max();
